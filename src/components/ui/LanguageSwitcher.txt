'use client';

import { usePathname, useRouter } from 'next/navigation';
import { locales } from '@/i18n/config';
import { createLocalizedPathname } from '@/i18n/i18n';

export function LanguageSwitcher() {
  const pathname = usePathname();
  const router = useRouter();
  
  const changeLanguage = (locale: string) => {
    const newPath = createLocalizedPathname(pathname, locale);
    router.push(newPath);
  };

  return (
    <div className="flex items-center space-x-2">
      {locales.map((locale) => (
        <button
          key={locale}
          onClick={() => changeLanguage(locale)}
          className="px-3 py-1 rounded hover:bg-gray-100 dark:hover:bg-gray-800 transition-colors"
        >
          {locale.toUpperCase()}
        </button>
      ))}
    </div>
  );
}
